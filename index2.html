<body>

  <div id="loading" class="loading-overlay"><h2>データを送信中...</h2></div>


  <div id="setup" class="card">

    <h2>(26)既存添削者フォーム③</h2>

    <p>(26)既存添削者フォーム①②に答えた情報を入力してください</p>

    <input type="text" id="email" placeholder="メールアドレス" style="width:90%; padding:10px; margin-bottom:10px;">

    <input type="text" id="id" placeholder="ID下6桁" style="width:90%; padding:10px; margin-bottom:10px;">

    <input type="text" id="name" placeholder="氏名" style="width:90%; padding:10px; margin-bottom:20px;">

    <button onclick="start()">認証して開始（記述回答）</button>

  </div>


  <div id="test" style="display:none;" class="card">

    <h3 id="sec-title">セクション1</h3>

    <textarea id="answer-box" placeholder="こちらに回答を記述してください..."></textarea>

    <br>

    <button id="next-btn" onclick="next()">次へ進む</button>

  </div>


  <div id="res" style="display:none;" class="card"></div>


  <script>

    // ★ここにGASのデプロイURL（execで終わるもの）を貼り付けてください

    const GAS_URL = "https://script.google.com/macros/s/AKfycbyDi8wbW51pJKVon4Alro-BljPun7lfcT8gdQ7fQI_3W7Ol8kgD2UNv2qEYhaNLJv2V9A/exec"; 


    let cur = 1, myRow, userInfo;


    // GASにデータを送る共通関数

    async function callGas(data) {

      const response = await fetch(GAS_URL, {

        method: "POST",

        body: JSON.stringify(data)

      });

      return await response.json();

    }


    async function start() {

      userInfo = { 

        email: document.getElementById('email').value.trim(), 

        id: document.getElementById('id').value.trim(), 

        name: document.getElementById('name').value.trim() 

      };

      if(!userInfo.email || !userInfo.id) return alert("メアドとIDを入力してください");


      document.getElementById('loading').style.display = 'block';

      try {

        // フォーム③として認証（内部でフォーム②の履歴と照合されます）

        const res = await callGas({ action: "register", userData: userInfo, formNum: 2 });

        document.getElementById('loading').style.display = 'none';


        if(!res.success) return alert(res.message);

        

        myRow = res.row;

        document.getElementById('setup').style.display = 'none';

        document.getElementById('test').style.display = 'block';

      } catch (e) {

        document.getElementById('loading').style.display = 'none';

        alert("サーバー通信エラーが発生しました。URLが正しいか、GASが「全員」に公開されているか確認してください。");

      }

    }


    async function next() {

      const text = document.getElementById('answer-box').value.trim();

      if(!text) return alert("回答を入力してください");


      document.getElementById('loading').style.display = 'block';

      try {

        await callGas({ 

          action: "saveText", 

          row: myRow, 

          sectionNum: cur, 

          text: text, 

          formNum: 2 

        });

        

        document.getElementById('loading').style.display = 'none';

        document.getElementById('answer-box').value = ""; // 入力欄をクリア


        if(cur < 5) {

          cur++;

          document.getElementById('sec-title').innerText = "セクション" + cur;

          if(cur === 5) document.getElementById('next-btn').innerText = "最終送信して採点";

        } else {

          finish();

        }

      } catch (e) {

        document.getElementById('loading').style.display = 'none';

        alert("送信に失敗しました。再度お試しください。");

      }

    }


    async function finish() {

      document.getElementById('test').style.display = 'none';

      document.getElementById('loading').innerHTML = "<h2>AI採点中...<br>少々お待ちください</h2>";

      document.getElementById('loading').style.display = 'block';

      

      try {

        const score = await callGas({ action: "grade", row: myRow, formNum: 2 });

        document.getElementById('loading').style.display = 'none';

        

        let h = `<h2>採点結果: ${score}点</h2>`;

        if (score >= 30) {

          h += `<div style="color:green; font-weight:bold;">【合格】</div><p>おめでとうございます。次は<b>(26)既存添削者フォーム④</b>にお答えください。</p>`;

        } else {

          h += `<div style="color:red; font-weight:bold;">【不合格】</div><p>合格点は30点以上です。各セクションに10個の誤りがあります。<br>再度このページを読み込み直して、最初から挑戦してください。</p>`;

        }

        document.getElementById('res').innerHTML = h;

        document.getElementById('res').style.display = 'block';

      } catch (e) {

        document.getElementById('loading').style.display = 'none';

        alert("採点処理でエラーが発生しました。スプレッドシートを確認してください。");

      }

    }

  </script>
