    <div id="section-ui">

      <h3 id="sec-title">セクション1</h3>

      <button onclick="next()">回答を終了して次へ</button>

  <div id="loading" style="display:none;"><h2>処理中...</h2></div>


  <script>

    // ★ここに自分のGASの「デプロイURL」を貼り付けてください

    const GAS_URL = "https://script.google.com/macros/s/AKfycbyDi8wbW51pJKVon4Alro-BljPun7lfcT8gdQ7fQI_3W7Ol8kgD2UNv2qEYhaNLJv2V9A/exec"; 


    let cur = 1, timer, myRow, userInfo, recorder, chunks = [], stream;


    async function callGas(data) {

      const response = await fetch(GAS_URL, {

        method: "POST",

        body: JSON.stringify(data)

      });

      return await response.json();

    }


    async function start() {

      userInfo = { email: document.getElementById('email').value.trim(), id: document.getElementById('id').value.trim(), name: document.getElementById('name').value.trim() };

      if(!userInfo.email || !userInfo.id) return alert("入力してください");


      try {

        // 先にマイクを確保（GitHub上ならここで許可ダイアログが出ます！）

        stream = await navigator.mediaDevices.getUserMedia({ audio: true });

        

        // GASでユーザー照合

        const res = await callGas({ action: "register", userData: userInfo, formNum: 1 });

        if(!res.success) return alert(res.message);

        

        myRow = res.row;

        document.getElementById('setup').style.display='none';

        document.getElementById('test').style.display='block';

        document.getElementById('bar-container').style.display='block';

        setupRecorder();

      } catch (e) {

        alert("マイクを許可してください。GitHub Pagesなら許可できるはずです。");

      }

    }


    function setupRecorder() {

      recorder = new MediaRecorder(stream);

      recorder.ondataavailable = e => chunks.push(e.data);

      const ctx = new AudioContext();

      const ana = ctx.createAnalyser();

      ctx.createMediaStreamSource(stream).connect(ana);

      const data = new Uint8Array(ana.frequencyBinCount);

      const upd = () => { ana.getByteFrequencyData(data); document.getElementById('bar').style.width = (data.reduce((a,b)=>a+b)/data.length)*2.5+"%"; requestAnimationFrame(upd); };

      upd();

      runTimer();

    }


    function runTimer() {

      chunks = []; recorder.start();

      let left = 150;

      clearInterval(timer);

      timer = setInterval(() => {

        document.getElementById('time').innerText = Math.floor(left/60)+":"+(left%60).toString().padStart(2,'0');

        if(left-- <= 0) { clearInterval(timer); next(true); }

      }, 1000);

    }


    async function next(auto=false) {

      if(!auto && !confirm("次へ進みますか？")) return;

      document.getElementById('loading').style.display = 'block';

      recorder.stop();

      recorder.onstop = async () => {

        const reader = new FileReader();

        reader.readAsDataURL(new Blob(chunks, {type:'audio/webm'}));

        reader.onloadend = async () => {

          await callGas({ action: "saveAudio", row: myRow, sectionNum: cur, audioBase64: reader.result, userInfo: userInfo, formNum: 1 });

          document.getElementById('loading').style.display = 'none';

          if(cur < 5) {

            cur++;

            document.getElementById('sec-title').innerText = "セクション" + cur;

            runTimer();

          } else {

            finish();

          }

        };

      };

    }


    async function finish() {

      stream.getTracks().forEach(t => t.stop());

      document.getElementById('test').style.display = 'none';

      document.getElementById('loading').innerHTML = "<h2>AI採点中...</h2>";

      document.getElementById('loading').style.display = 'block';

      

      const score = await callGas({ action: "grade", row: myRow, formNum: 1 });

      let h = `<h2>点数: ${score}点</h2>`;

      h += (score >= 30) ? "合格！フォーム④へ" : "不合格。フォーム③へ";

      document.getElementById('loading').innerHTML = h;

    }
